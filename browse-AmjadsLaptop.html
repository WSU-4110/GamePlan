Unreal: Health Component

C++ 

#include "HealthComponent.h"
#include "GameFramework/Actor.h"

UHealthComponent::UHealthComponent() {
    PrimaryComponentTick.bCanEverTick = false;
    MaxHealth = 100.f;
    CurrentHealth = MaxHealth;
}

void UHealthComponent::ApplyDamage(float Amount) {
    CurrentHealth = FMath::Clamp(CurrentHealth - Amount, 0.f, MaxHealth);
    if (CurrentHealth <= 0.f) {
        // broadcast death or handle actor destroy
        OnDeath.Broadcast();
    }
}

C++: Minimal ECS (Entity/Component)

C++ 

#include <memory>
#include <vector>

struct Component { virtual ~Component() = default; };

struct Transform : Component {
    float x{0}, y{0};
};

class Entity {
    std::vector<std::unique_ptr<Component>> comps;
public:
    template<class T, class... Args>
    T& add(Args&&... args) {
        auto c = std::make_unique<T>(std::forward<Args>(args)...);
        T& ref = *c;
        comps.push_back(std::move(c));
        return ref;
    }
};

C++: Event Bus (Observer)

C++ 

#include <functional>
#include <string>
#include <vector>

struct Event { std::string name; };

class EventBus {
    std::vector<std::function<void(const Event&)>> subs;
public:
    void subscribe(std::function<void(const Event&)> fn) { subs.push_back(std::move(fn)); }
    void publish(const Event& e) {
        for (auto& fn : subs) fn(e);
    }
};

C++: AI Finite State Machine

C++ 

enum class State { Idle, Chase, Attack };

struct AI {
    State s{State::Idle};
    float dist{0.f};

    void update() {
        switch (s) {
            case State::Idle:   if (dist < 10) s = State::Chase; break;
            case State::Chase:  if (dist < 2)  s = State::Attack; break;
            case State::Attack: if (dist > 3)  s = State::Chase;  break;
        }
    }
};

C++: AABB Collision

C++ 

struct AABB { float x,y,w,h; };

inline bool intersects(const AABB& a, const AABB& b) {
    return !(a.x + a.w < b.x || b.x + b.w < a.x ||
             a.y + a.h < b.y || b.y + b.h < a.y);
}

C++: RAII ScopeTimer

C++ 

#include <chrono>
#include <iostream>

class ScopeTimer {
    std::chrono::high_resolution_clock::time_point t0;
    const char* label;
public:
    ScopeTimer(const char* name)
        : t0(std::chrono::high_resolution_clock::now()), label(name) {}
    ~ScopeTimer() {
        auto t1 = std::chrono::high_resolution_clock::now();
        auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(t1 - t0).count();
        std::cout << label << " took " << ms << "ms\n";
    }
};

Unreal: Actor Spawning

C++ 

#include "GameFramework/Actor.h"

void ASpawner::SpawnEnemy() {
    if (!EnemyClass) return;

    FVector SpawnLocation = GetActorLocation() + FVector(100, 0, 0);
    FRotator SpawnRotation = FRotator::ZeroRotator;

    FActorSpawnParameters SpawnParams;
    SpawnParams.Owner = this;
    SpawnParams.SpawnCollisionHandlingOverride =
        ESpawnActorCollisionHandlingMethod::AdjustIfPossibleButAlwaysSpawn;

    AActor* Enemy = GetWorld()->SpawnActor<AActor>(
        EnemyClass,
        SpawnLocation,
        SpawnRotation,
        SpawnParams
    );
}

Unreal: Enhanced Input Binding

#include "EnhancedInputComponent.h"
#include "EnhancedInputSubsystems.h"

void APlayerCharacter::SetupPlayerInputComponent(UInputComponent* PlayerInputComponent) {
    Super::SetupPlayerInputComponent(PlayerInputComponent);

    if (UEnhancedInputComponent* EIC = Cast<UEnhancedInputComponent>(PlayerInputComponent)) {
        // Bind movement action
        EIC->BindAction(MoveAction, ETriggerEvent::Triggered,
                        this, &APlayerCharacter::Move);

        // Bind jump action
        EIC->BindAction(JumpAction, ETriggerEvent::Started,
                        this, &ACharacter::Jump);
    }
}

void APlayerCharacter::Move(const FInputActionValue& Value) {
    FVector2D MoveVector = Value.Get<FVector2D>();
    AddMovementInput(GetActorForwardVector(), MoveVector.Y);
    AddMovementInput(GetActorRightVector(), MoveVector.X);
}

Unreal: Weapon Component

C++ 

#include "WeaponComponent.h"
#include "Kismet/GameplayStatics.h"

void UWeaponComponent::Fire() {
    if (CurrentAmmo <= 0) {
        UE_LOG(LogTemp, Warning, TEXT("Out of ammo!"));
        return;
    }

    CurrentAmmo--;

    FVector Start = GetOwner()->GetActorLocation();
    FVector Forward = GetOwner()->GetActorForwardVector();
    FVector End = Start + (Forward * 5000.f);

    FHitResult Hit;
    FCollisionQueryParams Params;
    Params.AddIgnoredActor(GetOwner());

    if (GetWorld()->LineTraceSingleByChannel(Hit, Start, End,
        ECC_Visibility, Params)) {
        // Apply damage to hit actor
        if (AActor* HitActor = Hit.GetActor()) {
            UGameplayStatics::ApplyDamage(HitActor, Damage,
                nullptr, GetOwner(), nullptr);
        }
    }
}

Unreal: Animation Blueprint Helper

C++ 

#include "Animation/AnimInstance.h"
#include "GameFramework/CharacterMovementComponent.h"

void UCharacterAnimInstance::NativeUpdateAnimation(float DeltaSeconds) {
    Super::NativeUpdateAnimation(DeltaSeconds);

    APawn* Owner = TryGetPawnOwner();
    if (!Owner) return;

    // Calculate speed
    Speed = Owner->GetVelocity().Size();

    // Get movement component
    if (UCharacterMovementComponent* MovComp =
        Owner->FindComponentByClass<UCharacterMovementComponent>()) {
        bIsInAir = MovComp->IsFalling();
        bIsCrouching = MovComp->IsCrouching();
    }

    // Direction for strafe blendspace
    FVector Velocity = Owner->GetVelocity();
    Direction = CalculateDirection(Velocity, Owner->GetActorRotation());
}
